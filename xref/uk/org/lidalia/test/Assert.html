24      public static Matcher<Class<?>> isNotInstantiable() {
25          return both(aClassWhoseSuperClass(is(equalTo(Object.class))))
26                  .and(aClassWhoseSetOfConstructors(both(
27                          is(Assert.<List<Constructor<?>>>aCollectionWhoseLength(is(1))))

81      public static <T extends Member> Matcher<T> isAMemberWithModifier(final Modifier modifier) {
82          return new TypeSafeDiagnosingMatcher<T>() {
83  
84              @Override
85              protected boolean matchesSafely(final T item, final Description mismatchDescription) {
86                  final boolean matches = modifier.isTrueOf(item);
87                  if (!matches) {
88                      mismatchDescription.appendValue(item).appendText(" did not have modifier ").appendValue(modifier);
89                  }
90                  return matches;
91              }
92  
93              @Override
94              public void describeTo(final Description description) {
95                  description.appendText("is a member with modifier " + modifier);
96              }
97          };
98      }

110     private static Matcher<Constructor<?>> aConstructorWhoseThrownException(final Matcher<? extends Throwable> throwableMatcher) {
111         return new FeatureMatcher<Constructor<?>, Throwable>(
112                 throwableMatcher, "a constructor whose thrown exception", "'s thrown exception") {
113             @Override
114             protected Throwable featureValueOf(final Constructor<?> constructor) {
115                 try {
116                     constructor.setAccessible(true);
117                     constructor.newInstance();
118                     return null;
119                 } catch (InvocationTargetException e) {
120                     return e.getCause();
121                 } catch (Exception e) {
122                     return throwUnchecked(e, null);
123                 } finally {
124                     constructor.setAccessible(false);
125                 }
126             }
