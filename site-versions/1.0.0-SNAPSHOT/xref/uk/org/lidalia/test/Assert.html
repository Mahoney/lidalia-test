29      /**
30       * Asserts that a class is not instantiable - it exists only for its static members.
31       * <p>
32       * More precisely, asserts that the class:
33       * <ul>
34       *     <li>Has Object as its immediate superclass</li>
35       *     <li>Has only one constructor</li>
36       *     <li>That constructor is private</li>
37       *     <li>That constructor takes no arguments</li>
38       *     <li>That constructor will throw an {@link UnsupportedOperationException} with message "Not Instantiable" if it is
39       *     invoked via reflection.</li>
40       * </ul>
41       * <p>
42       * Usage:
43       * {@code assertThat(Values.class, isNotInstantiable());}
44       *
45       * @return a matcher that asserts that a class is not instantiable
46       */
47      public static Matcher<Class<?>> isNotInstantiable() {
48          return both(aClassWhoseSuperClass(is(equalTo(Object.class))))
49                  .and(aClassWhoseSetOfConstructors(both(
50                          is(Assert.<List<Constructor<?>>>aCollectionWhoseSize(is(1))))
51                          .and(is(Assert.<List<Constructor<?>>, Constructor<?>>aListWhoseElementAtIndex(0, both(
52                                  is(aConstructorWhoseParameterTypes(is(Assert.<List<Class<?>>>aCollectionWhoseSize(is(0))))))
53                                  .and(isAMemberWithModifier(Modifier.PRIVATE))
54                                  .and(aConstructorWhoseThrownException(both(
55                                          isA(UnsupportedOperationException.class))
56                                          .and(is(aThrowableWhoseMessage(is("Not instantiable")))))))))));
57      }

60       * Facilitates making an assertion about the superclass of a {@link Class}.
61       * <p>
62       * Usage:
63       * {@code assertThat(String.class, is(aClassWhoseSuperClass(is(Object.class))));}
64       *
65       * @param classMatcher the matcher that will be applied to the class's superclass
66       * @param <U> the type of the superclass of the class being matched
67       * @param <T> the type of the class being matched
68       * @return a matcher that will assert something about the superclass of a class
69       */
70      public static <U, T extends U> FeatureMatcher<Class<? extends T>, Class<? extends U>> aClassWhoseSuperClass(
71              final Matcher<? extends Class<? extends U>> classMatcher) {
72          return new FeatureMatcher<Class<? extends T>, Class<? extends U>>(
73                  classMatcher, "a Class whose super class", "'s super class") {
74              @Override
75              protected Class<? extends U> featureValueOf(final Class<? extends T> actual) {
76                  return (Class<? extends U>) actual.getSuperclass();
77              }
78          };
79      }

81      private static FeatureMatcher<Class<?>, List<Constructor<?>>> aClassWhoseSetOfConstructors(
82              final Matcher<List<Constructor<?>>> matcher) {
83          return new FeatureMatcher<Class<?>, List<Constructor<?>>>(
84                  matcher, "a Class whose set of constructors", "'s constructors") {
85              @Override
86              protected List<Constructor<?>> featureValueOf(final Class<?> actual) {
87                  final List<Constructor<?>> constructors = asList(actual.getDeclaredConstructors());
88                  Collections.sort(constructors, new Comparator<Constructor<?>>() {
89                      @Override
90                      public int compare(final Constructor<?> one, final Constructor<?> other) {
91                          return one.toString().compareTo(other.toString());
92                      }
93                  });
94                  return constructors;
95              }
96          };
97      }

119      * Facilitates making an assertion about the element at a given index of a {@link List}.
120      * <p>
121      * Usage:
122      * {@code assertThat(asList("a", "b", "c"), is(aListWhoseElementAtIndex(2, is("c"))));}
123      *
124      * @param index the index of the element in the list the matcher will be applied to
125      * @param matcher the matcher that will be applied to the element
126      * @param <T> the type of the List
127      * @param <E> the type of the elements in the List
128      * @return a matcher that will assert something about the element at the given index of a collection
129      */
130     public static <T extends List<? extends E>, E> Matcher<T> aListWhoseElementAtIndex(
131             final Integer index, final Matcher<E> matcher) {
132         return new FeatureMatcher<T, E>(matcher, "a List whose element at index " + index, "'s element at index " + index) {
133             @Override
134             protected E featureValueOf(final T actual) {
135                 if (actual.size() > index) {
136                     return actual.get(index);
137                 } else {
138                     throw new AssertionError(actual + " has no element at index " + index);
139                 }
140             }
141         };
142     }

144     /**
145      * Asserts that a given {@link Member} has a given {@link Modifier}.
146      * <p>
147      * Usage:
148      * {@code assertThat(Object.class.getMethod("toString"), isAMemberWithModifier(Modifier.PUBLIC));}
149      *
150      * @param modifier the modifier the member is expected to have
151      * @param <T> the type of the Member
152      * @return a matcher that will assert a member has a modifier
153      */
154     public static <T extends Member> Matcher<T> isAMemberWithModifier(final Modifier modifier) {
155         return new TypeSafeDiagnosingMatcher<T>() {
156 
157             @Override
158             protected boolean matchesSafely(final T item, final Description mismatchDescription) {
159                 final boolean matches = modifier.existsOn(item);
160                 if (!matches) {
161                     mismatchDescription.appendValue(item).appendText(" did not have modifier ").appendValue(modifier);
162                 }
163                 return matches;
164             }
165 
166             @Override
167             public void describeTo(final Description description) {
168                 description.appendText("is a member with modifier " + modifier);
169             }
170         };
171     }

183     private static Matcher<Constructor<?>> aConstructorWhoseThrownException(final Matcher<? extends Throwable> throwableMatcher) {
184         return new FeatureMatcher<Constructor<?>, Throwable>(
185                 throwableMatcher, "a constructor whose thrown exception", "'s thrown exception") {
186             @Override
187             protected Throwable featureValueOf(final Constructor<?> constructor) {
188                 try {
189                     constructor.setAccessible(true);
190                     constructor.newInstance();
191                     return null;
192                 } catch (InvocationTargetException e) {
193                     return e.getCause();
194                 } catch (Exception e) {
195                     return throwUnchecked(e, null);
196                 } finally {
197                     constructor.setAccessible(false);
198                 }
199             }
200         };
201     }
